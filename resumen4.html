def facturacionAbonado(estrato,impulsos):
    if estrato==1:
        tarifaBasica=10000
    elif estrato==2:
        tarifaBasica=15000
    elif estrato==3:
        tarifaBasica=30000
    elif estrato==4:
        tarifaBasica=50000
    else:
        tarifaBasica=65000
    valorImpulsos=impulsos*100
    valorPagar=tarifaBasica+valorImpulsos
    return valorPagar,tarifaBasica,valorImpulsos
